149 void RobotDyn4DigitDisplay::time(uint8_t hour, uint8_t minute, bool doubleDotsOn,
150  bool padHours)
151 {
152  if (padHours || (hour >= 10)) {
153  displayBuffer[0] = pgm_read_byte(&SEGMENT_DATA[hour / 10]);
154  } else {
155  displayBuffer[0] = 0;
156  }
157  displayBuffer[1] = pgm_read_byte(&SEGMENT_DATA[hour % 10]) | (doubleDotsOn ? 0x80 : 0x00);
158  displayBuffer[2] = pgm_read_byte(&SEGMENT_DATA[minute / 10]);
159  displayBuffer[3] = pgm_read_byte(&SEGMENT_DATA[minute % 10]);
160  writeData(0x00, displayBuffer, sizeof(displayBuffer));
161 }

170 void RobotDyn4DigitDisplay::dec(int value, uint8_t pad)
171 {
172  uint8_t digit;
173  bool negative = false;
174 
175  // Check if value fits on display
176  if ((value > 9999) || (value < -999)) {
177  overflow();
178  return;
179  }
180 
181  if (value < 0) {
182  // Make value positive
183  value *= -1;
184 
185  // Set negative flag
186  negative = true;
187  }
188 
189  // Clear buffer
190  memset(displayBuffer, 0x00, ROBOT_DYN_4DIGIT_DISPLAY_NUM_DIGITS);
191 
192  // Value 0..9999 with padding
193  for (uint8_t i = 0; i < ROBOT_DYN_4DIGIT_DISPLAY_NUM_DIGITS; i++) {
194  digit = (uint8_t)(ROBOT_DYN_4DIGIT_DISPLAY_NUM_DIGITS - 1 - i);
195  if (value == 0) {
196  if (pad > i) {
197  displayBuffer[digit] = pgm_read_byte(&SEGMENT_DATA[0]);
198  } else {
199  displayBuffer[digit] = 0;
200  }
201  } else {
202  displayBuffer[digit] = pgm_read_byte(&SEGMENT_DATA[value % 10]);
203  }
204 
205  value /= 10;
206 
207  if ((value == 0) && negative) {
208  displayBuffer[--digit] = SEGMENTS_MINUS;
209  break;
210  }
211  }
212 
213  // Write display segment data
214  writeData(0x00, displayBuffer, sizeof(displayBuffer));
215 }
